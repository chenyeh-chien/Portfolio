### 作業前準備
#### 1. 環境建置
- 由於是MIS只能使用離線環境, 所以需要先在Windows環境用WSL 2建置好linux環境跑docker:
[WSL 2 Installation](/blog/wsl-2-installation)

#### 2. 建立共享資料夾與WSL 2之間的路徑
- 第一步：在 Windows 重新對映一個代號
    - 打開 檔案總管。
    - 在本機（This PC）點擊右鍵 -> 連線網路磁碟機 (Map network drive)。
    - 磁碟機 (Drive): 下拉選一個空的，例如 X:。
    - 資料夾 (Folder): 輸入公司的路徑 \\server\share。
    - 勾選「登入時重新連線」，點擊完成並輸入帳密。

- 第二步：在 WSL 2 (Ubuntu) 裡面掛載這個新代號
    - 現在我們要把 Windows 的 X: 槽映射進 Linux：

```bash
# 1. 建立掛載點 (目錄名稱可以隨你自己取)
sudo mkdir -p /mnt/company_share

# 2. 使用 drvfs 掛載 X 槽 (不需要安裝 cifs-utils)
sudo mount -t drvfs X: /mnt/company_share
```

### ⭐ Black Duck API 
#### 1. 認證 (取得 Bearer Token)
在呼叫任何 API 之前，必須先用 API Token 換取臨時的 bearerToken。

```
POST /api/tokens/authenticate
```

Header:
```
Authorization: token <YOUR_API_TOKEN>
Accept: application/vnd.blackducksoftware.user-4+json
```

#### 2. 獲取用戶清單 (Get Users)
這是您用來與 LDAP 進行比對的核心 API。

```
GET /api/users
```

- 常用參數 (Query Parameters):
    - limit: 預設值通常較小（如 20），建議設為 limit=1000 以一次獲取更多帳號。
    - offset: 用於分頁處理（如果帳號數超過 1000）。

Header:
```
Authorization: Bearer <YOUR_BEARER_TOKEN>
Accept: application/vnd.blackducksoftware.user-4+json
```

Output:
```json
{
  "totalCount": 150,
  "items": [
    {
      "userName": "john.doe",
      "firstName": "John",
      "lastName": "Doe",
      "email": "john.doe@company.com",
      "active": true,
      "type": "EXTERNAL", // EXTERNAL 通常指 LDAP/SAML 帳號，INTERNAL 為本地帳號
      "_meta": {
        "href": "https://bd.company.com/api/users/83a7...", // 這是該用戶的唯一 ID URL
        "links": [...]
      }
    }
  ]
}
```

#### 3. 獲取使用者的群組清單
```
GET /api/users/{userId}/usergroups
```

#### 4. 將使用者從群組移除
```
DELETE /api/usergroups/{userGroupId}/users/{userId}
```

#### 5. 更新使用者為 Inactive
```
PUT /api/users/{userId}
```

Body:
```json
{
  "userName": "target_user",
  "firstName": "...",
  "lastName": "...",
  "email": "...",
  "active": false,  // 關鍵點
  "type": "EXTERNAL"
}
```

### ⭐ ldap-check-users.js (OA VM):
```javascript
// get BLACK DUCK access token
import dotenv from "dotenv";
import fs from "fs-extra";
import path from "path";
import { fileURLToPath } from "url";
dotenv.config();

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const SRM_URL = process.env.SRM_URL;
const BLACKDUCK_USERS_FILE = path.join(__dirname, "../static/blackduck-users.json");
const RETIRED_USERS_FILE = path.join(__dirname, "../static/retired-users.json");

async function getSRMBearerToken() {
  const response = await fetch(`${SRM_URL}/api/Token/Token/FetchLongToken`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      "user": "SSS",
      "PassKey": "56945615-4739-4441-8788-2f5f8028c9cc",
      "Ip": "",
      "Email": ""
    })
  });

  if (!response.ok) {
    throw new Error(`Failed to get bearer token: ${response.statusText}`);
  }

  const data = await response.json();
  return data.ReturnMessage;
}

async function getLDAPUserInfoByNameList(bearerToken, list) {
  const input = list.join(",");
  const response = await fetch(`${SRM_URL}/api/Internal/UserStatus/UserList`, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${bearerToken}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      "Input": input
    })
  });
  
  if (!response.ok) {
    throw new Error(`Failed to get LDAP user info: ${response.statusText}`);
  }

  const data = await response.json();
  return data.Result;
}

async function readBlackDuckUsersFromJSON() {
  const blackDuckUsers = await fs.readJson(BLACKDUCK_USERS_FILE);
  if (blackDuckUsers.length === 0) {
    throw new Error("Black Duck users list is empty");
  }

  return blackDuckUsers;
}

async function readAndWriteFile() {
  const blackDuckUsersInfo = await readBlackDuckUsersFromJSON();
  const blackDuckUsersMap = blackDuckUsersInfo.reduce((acc, curr) => {
    return {
      ...acc,
      [curr.userName]: curr
    }
  }, {})
  
  const blackDuckUsers = blackDuckUsersInfo.map(user => user.userName);
  const bearerToken = await getSRMBearerToken();
  const ldapUsers = await getLDAPUserInfoByNameList(bearerToken, blackDuckUsers);
  const retiredUsers = ldapUsers
    .filter(user => user.AccountStatus === "X")
    .map(user => blackDuckUsersMap[user.Input]);
    
  await fs.writeJson(RETIRED_USERS_FILE, retiredUsers, { spaces: 4 });
}

readAndWriteFile();
```

import dotenv from "dotenv";
import fs from "fs-extra";
import path from "path";

const API_TOKEN = process.env.BLACK_DUCK_API_TOKEN;
const BLACK_DUCK_URL = process.env.BLACK_DUCK_URL;
const BLACKDUCK_USERS_FILE = path.join(__dirname, "../static/blackduck-users.json");

async function getBlackDuckBearerToken() {
  const response = await fetch(`${BLACK_DUCK_URL}/api/tokens/authenticate`, {
    method: "POST",
    headers: {
      "Authorization": `token ${API_TOKEN}`,
      "Accept": "application/vnd.blackducksoftware.user-4+json"
    }
  });

  if (!response.ok) {
    throw new Error(`Failed to get bearer token: ${response.statusText}`);
  }

  const data = await response.json();
  return data.bearerToken;
}

async function getAllUsers(bearerToken) {
  const response = await fetch(`${BLACK_DUCK_URL}/api/users?limit=1000`, {
    method: "GET",
    headers: {
      "Authorization": `Bearer ${bearerToken}`,
      "Accept": "application/vnd.blackducksoftware.user-4+json"
    }
  });

  if (!response.ok) {
    throw new Error(`Failed to get all users: ${response.statusText}`);
  }

  const data = await response.json();
  return data.items;
}

async function writeBlackDuckUsers() {
  const blackDuckBearerToken = await getBlackDuckBearerToken();
  const allUsers = await getAllUsers(blackDuckBearerToken);
  const filteredUsers = 
	allUsers.filter(user => 
		user.active && 
		(
			user.userName !== 'sysadmin' && 
			user.userName !== 'pega_sentinel'
		)
	);
  await fs.writeJson(BLACKDUCK_USERS_FILE, filteredUsers, { spaces: 4 });
}

writeBlackDuckUsers();

### ⭐ blackduck-remove-users.js (Test VM):
```javascript
import dotenv from "dotenv";
import fs from "fs-extra";
import path from "path";
import { fileURLToPath } from "url";
dotenv.config();

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const API_TOKEN = process.env.BLACK_DUCK_API_TOKEN;
const BLACK_DUCK_URL = process.env.BLACK_DUCK_URL;
const RETIRED_USERS_FILE = path.join(__dirname, "../static/retired-users.json");

async function getBlackDuckBearerToken() {
  const response = await fetch(`${BLACK_DUCK_URL}/api/tokens/authenticate`, {
    method: "POST",
    headers: {
      "Authorization": `token ${API_TOKEN}`,
      "Accept": "application/vnd.blackducksoftware.user-4+json"
    }
  });

  if (!response.ok) {
    throw new Error(`Failed to get bearer token: ${response.statusText}`);
  }
  
  const data = await response.json();
  return data.bearerToken;
}

async function getUserGroupsByUserId(bearerToken, userId) {
  const response = await fetch(`${BLACK_DUCK_URL}/api/users/${userId}/usergroups`, {
    method: "GET",
    headers: {
      "Authorization": `Bearer ${bearerToken}`,
      "Accept": "application/vnd.blackducksoftware.user-4+json"
    }
  });

  if (!response.ok) {
	console.log(response)
    throw new Error(`Failed to get user groups: ${response.statusText}`);
  }

  return await response.json();
}

async function removeUserFromGroup(
    bearerToken, 
    userGroupId, 
    userId
) {
  const response = 
    await fetch(`${BLACK_DUCK_URL}/api/usergroups/${userGroupId}/users/${userId}`, {
        method: "DELETE",
        headers: {
            "Authorization": `Bearer ${bearerToken}`,
            "Accept": "application/vnd.blackducksoftware.user-4+json"
        }
    });

  if (!response.ok) {
	console.log(response)
    throw new Error(`Failed to remove user from group: ${response.statusText}`);
  }
}

async function updateUser(
    bearerToken, 
    userId, 
    data
) {
  const response = 
    await fetch(`${BLACK_DUCK_URL}/api/users/${userId}`, {
        method: "PUT",
        headers: {
            "Authorization": `Bearer ${bearerToken}`,
            "Content-Type": "application/json"
        },
        body: JSON.stringify(data)
    });

  if (!response.ok) {
    throw new Error(`Failed to update user: ${response.statusText}`);
  }
}

async function readRetiredUsersFromJSON() {
  const retiredUsers = await fs.readJson(RETIRED_USERS_FILE);
  if (retiredUsers.length === 0) {
    throw new Error("Retired users list is empty");
  }

  return retiredUsers;
}

function getIdByHref(href) {
  if (
	typeof href !== 'string' ||
	href.length === 0
  ) {
	throw new Error('Failed to get user id by href');  
  }
  
  const list = href.split("/");
  return list[list.length - 1];
}

async function deactivateUsers() {
  // get bearer tokens
  const retiredUsers = await readRetiredUsersFromJSON();
  const blackDuckBearerToken = await getBlackDuckBearerToken();
  
  for (let user of retiredUsers) {
	if (!user._meta || !user._meta.href) continue;
	
	const href = user._meta.href;
	const userId = getIdByHref(href);
    // get user groups
    const userGroups = 
        await getUserGroupsByUserId(blackDuckBearerToken, userId);
	
	const items = userGroups.items;
    // remove user from groups
    for (let group of items) {
	  if (!group.userGroup) continue;

	  const usergroupId = getIdByHref(group.userGroup);
      await removeUserFromGroup(blackDuckBearerToken, usergroupId, userId);
    }
    
    // deactivate the user
    await updateUser(blackDuckBearerToken, userId,{
      ...user,
      active: false
    });
  }
}

deactivateUsers();
```
